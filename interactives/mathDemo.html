Interactive Quadratic Function Demo

Interactive Quadratic Function Demo#

Use the sliders below to explore how the parameters of a quadratic equation affect its graph: [ y = ax^2 + bx + c ]

import numpy as np
import matplotlib.pyplot as plt
from ipywidgets import interact, FloatSlider
import warnings
warnings.filterwarnings("ignore")
plt.style.use("seaborn-v0_8-whitegrid")
def plot_quadratic(a=1.0, b=0.0, c=0.0):
    x = np.linspace(-10, 10, 400)
    y = a * x**2 + b * x + c

    plt.figure(figsize=(8, 5))
    plt.plot(x, y, color = "blue", linewidth = 2)
    plt.axhline(0, color='black', lw=1)
    plt.axvline(0, color='black', lw=1)
    plt.title(f'Quadratic Function: y = {a}x + {b}x + {c}')
    plt.xlabel('x')
    plt.ylabel('y')
    plt.grid(True)
    plt.ylim(-10, 10)
    plt.show()

interact(
    plot_quadratic,
    a=FloatSlider(value=1, min=-5, max=5, step=0.1, description= ''),
    b=FloatSlider(value=0, min=-5, max=5, step=0.5, description=r''),
    c=FloatSlider(value=0, min=-5, max=5, step=0.5, description=r'')
)
